import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
# Define the years and market shares
years = [2018, 2019, 2020, 2021, 2022]
market_share = {
    'Samsung': [35, 55, 50, 45, 45],
    'Xiaomi': [30, 25, 25, 40, 30],
    'OnePlus': [5, 5, 10, 10, 5]
}
# Create a DataFrame
df = pd.DataFrame(market_share, index=years)
# Calculate moving averages
moving_avg = df.rolling(window=2).mean()
# Simulated error values
np.random.seed(0)
errors = {
    'Samsung': np.random.uniform(1, 3, size=5),
    'Xiaomi': np.random.uniform(1, 3, size=5),
    'OnePlus': np.random.uniform(1, 3, size=5)
}
# Plotting the time series
plt.figure(figsize=(12, 6), facecolor='lightgrey')
colors = ['#1f77b4', '#ff7f0e', '#2ca02c']
linestyles = ['-', '--', '-.']
for i, brand in enumerate(df.columns):
    plt.plot(df.index, df[brand], marker='o', label=brand, color=colors[i], linestyle=linestyles[i], linewidth=2, markersize=8)
    plt.plot(moving_avg.index, moving_avg[brand], linestyle=':', color=colors[i], linewidth=1, alpha=0.7, label=f'{brand} Moving Avg')
    plt.errorbar(df.index, df[brand], yerr=errors[brand], fmt='o', color=colors[i], capsize=3, alpha=0.5)
    for year, sales in zip(df.index, df[brand]):
        plt.text(year, sales + 1, f"{int(sales)}%", ha='center', va='bottom', fontsize=10, fontweight='bold', color=colors[i])
    max_value = df[brand].max()
    max_year = df.index[np.argmax(df[brand])]
    plt.scatter(max_year, max_value, color='red', s=100, zorder=5)
    plt.text(max_year, max_value - 3, f'Peak: {max_year}', ha='center', va='top', fontsize=9, color='black')
# Title and labels
plt.title('Mobile Phone Market Share (2018-2022)', fontsize=18, fontweight='bold')
plt.xlabel('Year', fontsize=14)
plt.ylabel('Market Share (%)', fontsize=14)
plt.xticks(df.index)
plt.yticks(np.arange(0, 101, 5))
plt.grid(alpha=0.3)
# Percentage change calculation
offset = 0.04
for brand in df.columns:
    for year in range(1, len(df)):
        change = df[brand].iloc[year] - df[brand].iloc[year - 1]
        color = 'green' if change > 0 else 'red'
        plt.text(df.index[year] + offset, df[brand].iloc[year] + 2, f"{change:+.1f}%", ha='left', va='center', fontsize=9, color=color)
# Legend
plt.legend(title='Brands', fontsize=12, loc='upper right')
plt.tight_layout()
# Save the plot to a file
plt.savefig('enhanced_mobile_market_share_with_moving_avg_and_changes.png', dpi=300)
plt.show()
